4. Create a login application using Django with validation and error handling
Answer →

views.py
from django.shortcuts import render
def login(request):
    if request.method == "POST":
        u = request.POST["username"]
        p = request.POST["password"]
        if u == "admin" and p == "admin123":
            return render(request, "success.html")
        else:
            return render(request, "login.html", {"error": "Invalid Credentials"})
    return render(request, "login.html")

1. Design Django form to accept student details with validation
Answer →
forms.py

from django import forms
class StudentForm(forms.Form):
    fname = forms.CharField()
    lname = forms.CharField()
    student_class = 
forms.ChoiceField(choices=[("FY","FY"),("SY","SY"),("TY","TY")])
    email = forms.EmailField()
    mobno = forms.RegexField(regex=r'^\d{10}$')
    address = forms.CharField(widget=forms.Textarea)
    resume = forms.FileField()

views.py

from django.shortcuts import render
from .forms import StudentForm
def student(request):
    if request.method == "POST":
        form = StudentForm(request.POST, request.FILES)
        if form.is_valid():
            return render(request, "display.html", {"data": form.cleaned_data})
    else:
        form = StudentForm()
    return render(request, "form.html", {"form": form})
templates/form.html
<form method="post" enctype="multipart/form-data">
{% csrf_token %}
{{ form }}
<button>Submit</button>
</form>

templates/display.html
